# Helper Functions
@st.cache_data(ttl=300)
def extract_action_value(actions, action_type):
    """Extract specific action values"""
    if not actions:
        return 0
    for action in actions:
        if action.get('action_type') == action_type:
            return float(action.get('value', 0))
    return 0

@st.cache_data(ttl=300)
def fetch_data(endpoint, params):
    """Generic data fetching function with caching"""
    url = f"{BASE_URL}/{endpoint}"
    params['access_token'] = ACCESS_TOKEN
    
    try:
        response = requests.get(url, params=params)
        response.raise_for_status()
        return response.json()
    except Exception as e:
        st.error(f"Error fetching data: {str(e)}")
        return {"data": []}

def get_campaigns_performance(start_date, end_date):
    """Fetch campaign-level performance"""
    params = {
        'level': 'campaign',
        'time_range': json.dumps({
            'since': start_date,
            'until': end_date
        }),
        'fields': 'campaign_id,campaign_name,spend,impressions,clicks,ctr,cpm,cpc,actions,action_values,purchase_roas',
        'limit': 100
    }
    return fetch_data(f"{AD_ACCOUNT_ID}/insights", params)

def process_campaign_data(campaigns_data, status_data):
    """Process campaign data with ROAS flagging"""
    campaigns = []
    status_dict = {item['id']: item for item in status_data.get('data', [])}
    
    for campaign in campaigns_data.get('data', []):
        campaign_id = campaign.get('campaign_id')
        spend = float(campaign.get('spend', 0))
        
        purchases = extract_action_value(campaign.get('actions'), 'purchase')
        revenue = extract_action_value(campaign.get('action_values'), 'purchase')
        
        roas = revenue / spend if spend > 0 else 0
        
        status_info = status_dict.get(campaign_id, {})
        status = status_info.get('effective_status', 'UNKNOWN')
        
        campaigns.append({
            'Campaign Name': campaign.get('campaign_name'),
            'Status': status,
            'Spend': spend,
            'Revenue': revenue,
            'ROAS': roas,
            'Purchases': int(purchases),
            'CPM': float(campaign.get('cpm', 0)),
            'CTR': float(campaign.get('ctr', 0)),
            'CPC': float(campaign.get('cpc', 0)),
            'Impressions': int(campaign.get('impressions', 0)),
            'Clicks': int(campaign.get('clicks', 0)),
            'ROAS_Flag': '🟢 WINNER' if roas >= 5 else '🔴 LOSER'
        })
    
    return pd.DataFrame(campaigns)

def process_creative_data(creative_data):
    """Process creative performance"""
    creatives = []
    
    for ad in creative_data.get('data', []):
        spend = float(ad.get('spend', 0))
        purchases = extract_action_value(ad.get('actions'), 'purchase')
        revenue = extract_action_value(ad.get('action_values'), 'purchase')
        
        roas = revenue / spend if spend > 0 else 0
        
        creatives.append({
            'Ad Name': ad.get('ad_name'),
            'Campaign': ad.get('campaign_name'),
            'Adset': ad.get('adset_name'),
            'Impressions': int(ad.get('impressions', 0)),
            'Clicks': int(ad.get('clicks', 0)),
            'CTR': float(ad.get('ctr', 0)),
            'Spend': spend,
            'Purchases': int(purchases),
            'Revenue': revenue,
            'ROAS': roas,
            'CPM': float(ad.get('cpm', 0)),
            'CPC': float(ad.get('cpc', 0))
        })
    
    df = pd.DataFrame(creatives)
    
    if not df.empty:
        median_roas = df['ROAS'].median()
        median_ctr = df['CTR'].median()
        
        def classify_creative(row):
            if row['ROAS'] >= 5 and row['CTR'] >= median_ctr:
                return '🏆 TOP WINNER'
            elif row['ROAS'] >= 5:
                return '🟢 WINNER'
            elif row['ROAS'] < 2:
                return '🔴 LOSER'
            else:
                return '🟡 AVERAGE'
        
        df['Performance'] = df.apply(classify_creative, axis=1)
    
    return df

def process_demographic_data(demographic_data):
    """Process demographic breakdown"""
    demographics = {}
    
    for item in demographic_data.get('data', []):
        age = item.get('age', 'Unknown')
        gender = item.get('gender', 'Unknown')
        demo_key = f"{age} - {gender}"
        
        spend = float(item.get('spend', 0))
        purchases = extract_action_value(item.get('actions'), 'purchase')
        revenue = extract_action_value(item.get('action_values'), 'purchase')
        
        if demo_key not in demographics:
            demographics[demo_key] = {
                'Spend': 0, 'Impressions': 0, 'Clicks': 0,
                'Purchases': 0, 'Revenue': 0
            }
        
        demographics[demo_key]['Spend'] += spend
        demographics[demo_key]['Impressions'] += int(item.get('impressions', 0))
        demographics[demo_key]['Clicks'] += int(item.get('clicks', 0))
        demographics[demo_key]['Purchases'] += purchases
        demographics[demo_key]['Revenue'] += revenue
    
    demo_list = []
    for demo, data in demographics.items():
        roas = data['Revenue'] / data['Spend'] if data['Spend'] > 0 else 0
        ctr = (data['Clicks'] / data['Impressions'] * 100) if data['Impressions'] > 0 else 0
        
        demo_list.append({
            'Demographic': demo,
            'Spend': data['Spend'],
            'Impressions': data['Impressions'],
            'Clicks': data['Clicks'],
            'CTR': ctr,
            'Purchases': int(data['Purchases']),
            'Revenue': data['Revenue'],
            'ROAS': roas,
            'Flag': '🟢' if roas >= 5 else '🔴'
        })
    
    return pd.DataFrame(demo_list)
